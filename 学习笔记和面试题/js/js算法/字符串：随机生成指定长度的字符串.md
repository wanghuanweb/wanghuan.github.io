function randomString(n) {

    let str = "abcdefghijklmnopqrstuvwxyz9876543210",
        len = str.length,
        result = "";

    for(let i = 0;i < n;i++) {
        result += str.charAt(Math.floor(Math.random()*len));
    }

    return result;
}
