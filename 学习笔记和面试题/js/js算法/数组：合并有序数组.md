将两个有序数组合并为一个数组。请不要使用concat以及sort方法。

```
function merge(left, right) {
  var result = [],
    il = 0,
    ir = 0;

  while (il < left.length && ir < right.length) {
    if (left[il] < right[ir]) {
      result.push(left[il++]);
    } else {
      result.push(right[ir++]);
    }
  }
  return result.concat(left.slice(il)) .concat(right.slice(ir));
}
var left = [1, 4,7,8,10];
var right = [2, 5,47,90];
console.log(merge(left, right))
```
