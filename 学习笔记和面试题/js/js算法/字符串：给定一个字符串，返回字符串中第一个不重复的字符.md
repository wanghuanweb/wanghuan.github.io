字符串：给定一个字符串，返回字符串中第一个不重复的字符
,如abcbad的答案是 c

```
function firstFdupChar(str){
    let len = str.length,
        obj = {};

    if(len == 1) {
        return str;
    }

    for(let i = 0;i < len;i++) {
        let char = str.charAt(i);

        if(!obj[char]) {
            obj[char] = 1;
        } else {
            obj[char] += 1;
        }
    }

    for(let i in obj) {
        if(obj[i] == 1) {
            return i;
        }
    }
}
```
